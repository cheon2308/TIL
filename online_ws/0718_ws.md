# 3. 주민번호 비식별 처리

```python
num = list(str(input()))

i=0
id=''
for i in range(0 ,len(num)): #주민번호 길이만큼 반복

    if i <= 5:

        id = str(id) + num[i] # 생년월일은 출력

    else:
        id = str(id) + '*' #그 외에는 비식별
    i = i+1

print(id)
```

# 4. 2와 7의 배수의 합 구하기

```python
number = list(range(1,1000))

A =0

for i in number:
    if i % 2 == 0:
        A = A + i
    elif i % 7 == 0:
        A = A + i
    elif i % 2 and i % 7 == 0:
        A = A - i
    else:
        continue
    i+= 1

print(A)
```
