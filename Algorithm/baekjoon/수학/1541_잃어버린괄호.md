[백준 1541_잃어버린 괄호](https://www.acmicpc.net/problem/1541)

```python
N= input()

number = [i for i in range(10)]
a = ''
modify = []
# 우선 숫자를 int형으로 바꿔준다.
for i in N:
    # 수식이 아니라면 숫자를 만들어준다.
    if i != '+' and i != '-':
        a += i
    else:
        # 수식인 경우 새로운 수식 리스트에 추가해준다.
        modify.append(int(a))
        modify.append(i)
        a = ''
    # 마지막까지 왔다면 숫자 추가해주고 종료
    if i == N[-1]:
        modify.append(int(a))


# print(type(modify[0]))
# - 개수 체크
cnt = 0
minus = 0
for i in modify:
    if i == '-':
        minus +=1


for i in range(len(modify)):
    if modify[i] == '-':
        cnt +=1
        # 첫 -라면 뒤에 (를 붙여주고
        if cnt == 1:
            modify[i]='-('
        # 두번째 -라면 앞에 )를 붙여준다.
        elif cnt == 2:
            modify[i] =')-'
            cnt = 0
    # 만약 -가 홀수개로 나온다면 괄호가 닫히지 않으므로 마지막에 붙여준다.
    if minus > 0 and i == len(modify)-1:
        modify.append(')')

sum1 = 0
for k in modify:
    sum1 +=k
print(sum1)
```
